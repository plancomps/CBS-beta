  \KEY{Rule} ~ 
    & \SEMREF{eval} \LEFTPHRASE ~ \LEX{let} ~ \VARREF{X} ~ \LEX{={}} ~ \VARREF{E}\SUB{1} ~ \LEX{in} ~ \VARREF{E}\SUB{2} ~ \RIGHTPHRASE  = \\&\quad
